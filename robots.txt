# y facilita el sitemap para mejorar el SEO.

# Bloqueo de carpetas internas que no deben indexarse
Disallow: /assets/
Disallow: /images/
Disallow: /css/
Disallow: /js/
Disallow: /vendor/
Disallow: /node_modules/
Disallow: /drafts/       # por si usas posts en borrador

# Sitemap generado automáticamente por jekyll-sitemap

# Extra: indicamos que no queremos indexar páginas de búsqueda
Disallow: /search

# Googlebot: permitimos que use recursos JS/CSS (mejora indexación)

Allow: /.css

# Bingbot: igual que Google
User-agent: Bingbot
