# Lavadora Interactiva Fácil

Por favor, pasa el ratón o toca sobre cada parte de la lavadora para saber qué es y ver el enlace al recambio.

---

  <!-- Cuerpo de lavadora -->

<div id="info" style="font-size: 1.3em; padding: 15px; border: 2px solid #004d40; background: #b2dfdb; color: #004d40; min-height: 80px; max-width: 400px;">
  <strong>Pasa el ratón o toca sobre una parte para ver información aquí.</strong>
</div>

<script>
  const info = document.getElementById('info');
  const partes = {
    puerta: {
      nombre: "Puerta frontal",
      enlace: "/recambios/puerta.html"
    },
    panel: {
      nombre: "Panel de control",
      enlace: "/recambios/panel.html"
    },
    cajetin: {
      nombre: "Cajetín de detergente",
      enlace: "/recambios/cajetin.html"
    }
  };

  document.querySelectorAll('.parte').forEach(el => {
    el.addEventListener('mouseenter', () => {
      const parte = partes[el.id];
      if (parte) {
        info.innerHTML = `
          <div><strong>${parte.nombre}</strong></div>
          <div><a href="${parte.enlace}" target="_blank" style="color:#004d40; text-decoration: underline;">Ver recambio</a></div>
        `;
      }
    });
    el.addEventListener('mouseleave', () => {
      info.innerHTML = '<strong>Pasa el ratón o toca sobre una parte para ver información aquí.</strong>';
    });
  });
</script>
