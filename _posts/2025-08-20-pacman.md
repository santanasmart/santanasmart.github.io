  <script>
  (()=>{'use strict';

  // ====== Parámetros del juego ======
  const TILE = 18;              // tamaño del tile en píxeles (render nativo)
  const COLS = 38;              // columnas del grid
  const ROWS = 46;              // filas del grid
  const SPEED_PAC = 8.5;        // tiles por segundo
  const SPEED_GHOST = 7.5;      // tiles por segundo (modo normal)
  const SPEED_FRIGHT = 5.0;     // tiles por segundo (fantasma asustado)
  const FRIGHT_TIME = 6.0;      // segundos de vulnerabilidad
  const GHOST_COLORS = ['#ff4b4b','#ff78ff','#4bf1ff','#ffb84b'];
  const POWER_POS = [ // se colocan 4 power-dots en esquinas accesibles
    [2,2],[COLS-3,2],[2,ROWS-3],[COLS-3,ROWS-3]
  ];

  // Canvas y HUD
  const cvs = document.getElementById('game');
  const ctx = cvs.getContext('2d');
  const scoreEl = document.getElementById('score');
  const levelEl = document.getElementById('level');
  const livesEl = document.getElementById('lives');
  const overlayMsg = document.getElementById('overlayMsg');
  const stateTitle = document.getElementById('stateTitle');
  const stateSubtitle = document.getElementById('stateSubtitle');

  // Ajuste de tamaño del canvas según grid
  cvs.width  = COLS * TILE;
  cvs.height = ROWS * TILE;

  // ====== Utilidades ======
  const clamp=(v,min,max)=>v<min?min:(v>max?max:v);
  const rand=(a,b)=>Math.random()*(b-a)+a;
  const choice = arr => arr[(Math.random()*arr.length)|0];
  const keyset = new Set();

  // Vectores de dirección
  const DIRS = {
    'left':  {x:-1,y: 0},
    'right': {x: 1,y: 0},
    'up':    {x: 0,y:-1},
    'down':  {x: 0,y: 1},
    'none':  {x: 0,y: 0}
  };
  const OPP = {left:'right',right:'left',up:'down',down:'up'};

  // ====== Generación del laberinto ======
  // 0 vacío, 1 muro, 2 punto, 3 power, (puertas no implementadas para simplificar)
  const grid = Array.from({length:ROWS},()=>Array(COLS).fill(0));

  function rect(x,y,w,h) {
    for (let r=y;r<y+h;r++) for (let c=x;c<c+w;c++) {
      if (r>=0 && r<ROWS && c>=0 && c<COLS) grid[r][c]=1;
    }
  }

  // Bordes
  rect(0,0,COLS,1); rect(0,ROWS-1,COLS,1); rect(0,0,1,ROWS); rect(COLS-1,0,1,ROWS);

  // Túneles laterales: abrimos huecos en la pared izquierda/derecha en fila media
  const TUNNEL_ROW = (ROWS/2)|0;
  grid[TUNNEL_ROW][0]=0; grid[TUNNEL_ROW][COLS-1]=0;

  // Bloques internos simétricos (simple, estilo arena con casa central)
  function blocks(){
    // columnas gruesas
    rect(6,4,2,12); rect(COLS-8,4,2,12);
    rect(6,ROWS-16,2,12); rect(COLS-8,ROWS-16,2,12);
    // barras horizontales
    rect(3,9,12,2); rect(COLS-15,9,12,2);
    rect(3,ROWS-11,12,2); rect(COLS-15,ROWS-11,12,2);
    // casa central de fantasmas
    rect((COLS/2|0)-5,(ROWS/2|0)-2,10,5);
    // puerta de la casa (abrimos hueco)
    for (let c=(COLS/2|0)-1;c<=(COLS/2|0)+1;c++) grid[(ROWS/2|0)-2][c]=0;
  }
  blocks();

  // Colocación de puntos en todas las celdas transitables
  for (let r=0;r<ROWS;r++){
    for (let c=0;c<COLS;c++){
      if (grid[r][c]===0) grid[r][c]=2; // punto
    }
  }
  // Muros se mantienen como 1, power se pondrá a mano:
  POWER_POS.forEach(([c,r])=>{ if (grid[r]?.[c]===2) grid[r][c]=3; });

  // Posiciones iniciales
  const PAC_START = { c: (COLS/2|0), r: (ROWS-6) };
  const GHOST_HOME = { c: (COLS/2|0), r: (ROWS/2|0) };

  // No queremos puntos en las posiciones de inicio
  grid[PAC_START.r][PAC_START.c] = 0;
  grid[GHOST_HOME.r][GHOST_HOME.c] = 0;

  // Contar puntos totales
  function countDots(){
    let dots=0;
    for (let r=0;r<ROWS;r++) for (let c=0;c<COLS;c++) if (grid[r][c]===2||grid[r][c]===3) dots++;
    return dots;
  }

  // ====== Entidades ======
  class Entity {
    constructor(c,r,color){
      this.x = c*TILE+TILE/2; // posición en px (centro de celda)
      this.y = r*TILE+TILE/2;
      this.dir = 'none';
      this.nextDir = 'none';
      this.color = color||'#fff';
      this.radius = TILE*0.48;
      this.speed = SPEED_PAC;
    }
    tileC(){ return Math.floor(this.x / TILE); }
    tileR(){ return Math.floor(this.y / TILE); }
    atCenter(){
      // ¿está cerca del centro de su celda?
      const cx = this.tileC()*TILE + TILE/2;
      const cy = this.tileR()*TILE + TILE/2;
      return Math.abs(this.x - cx) < 0.5 && Math.abs(this.y - cy) < 0.5;
    }
    trySetDir(d){
      // Cambiar dirección solo si la celda siguiente es transitable
      const v = DIRS[d];
      const nc = this.tileC()+v.x;
      const nr = this.tileR()+v.y;
      if (!isWall(nc,nr)) { this.dir = d; return true; }
      return false;
    }
    move(dt, speed){
      const v = DIRS[this.dir];
      let nx = this.x + v.x * speed * TILE * dt;
      let ny = this.y + v.y * speed * TILE * dt;

      // Warp en túnel
      if (this.tileR()===TUNNEL_ROW){
        if (nx < -TILE/2) nx = (COLS*TILE)-TILE/2;
        if (nx > (COLS*TILE)+TILE/2) nx = TILE/2;
      }

      // Colisión con muros: aproximamos encerrando al centro si chocaría
      const nextC = Math.floor((nx)/TILE);
      const nextR = Math.floor((ny)/TILE);
      if (!isWall(nextC,nextR)) { this.x = nx; this.y = ny; }
      else {
        // pegamos al centro del tile actual para evitar temblores
        this.x = this.tileC()*TILE + TILE/2;
        this.y = this.tileR()*TILE + TILE/2;
        this.dir = 'none';
      }
    }
  }

  class Pacman extends Entity {
    constructor(){ super(PAC_START.c,PAC_START.r,'#ffd84a'); this.mouth=0; this.alive=true; }
    update(dt){
      // Al llegar al centro, intentamos aplicar la próxima dirección
      if (this.atCenter() && this.nextDir!=='none'){
        this.trySetDir(this.nextDir);
      }
      this.move(dt, SPEED_PAC);
      this.mouth = (this.mouth + dt*8) % (Math.PI*2);
      // Comer punto
      const c=this.tileC(), r=this.tileR();
      if (grid[r]?.[c]===2){ grid[r][c]=0; score+=10; dotsLeft--; }
      else if (grid[r]?.[c]===3){ grid[r][c]=0; score+=50; dotsLeft--; frightTimer=FRIGHT_TIME; ghosts.forEach(g=>g.setFright()); }
    }
    draw(){
      const ang = Math.abs(Math.sin(this.mouth))*0.6; // animación simple
      const v = DIRS[this.dir];
      let rot = 0;
      if (this.dir==='right') rot=0;
      else if (this.dir==='left') rot=Math.PI;
      else if (this.dir==='up') rot=-Math.PI/2;
      else if (this.dir==='down') rot=Math.PI/2;

      ctx.save();
      ctx.translate(this.x,this.y);
      ctx.rotate(rot);
      ctx.fillStyle=this.color;
      ctx.beginPath();
      ctx.moveTo(0,0);
      ctx.arc(0,0,this.radius, ang, Math.PI*2-ang);
      ctx.closePath();
      ctx.fill();
      ctx.restore();
    }
  }

  class Ghost extends Entity {
    constructor(i){
      super(GHOST_HOME.c + (i-1)-1, GHOST_HOME.r, GHOST_COLORS[i%GHOST_COLORS.length]);
      this.dir = choice(['left','right','up','down']);
      this.mode = 'chase';  // 'chase' | 'fright' | 'eaten'
      this.eyesOnly = false; // dibujar ojos cuando 'eaten'
      this.scatterTarget = {c: i%2===0 ? 1 : COLS-2, r: i<2 ? 1 : ROWS-2}; // esquinas
    }
    speed(){
      if (this.mode==='fright') return SPEED_FRIGHT;
      if (this.mode==='eaten')  return SPEED_GHOST*1.2;
      return SPEED_GHOST;
    }
    setFright(){
      if (this.mode==='eaten') return; // si está comido, ignora
      this.mode='fright';
      this.dir = OPP[this.dir] || this.dir;
    }
    setEaten(){
      this.mode='eaten';
      this.eyesOnly=true;
    }
    setChase(){ this.mode='chase'; this.eyesOnly=false; }
    homeReached(){
      return this.tileC()===GHOST_HOME.c && this.tileR()===GHOST_HOME.r;
    }
    chooseDir(){
      // En el centro del tile elegimos dirección
      if (!this.atCenter()) return;
      const options = ['left','right','up','down'].filter(d=>{
        if (OPP[d]===this.dir) return false; // evitar reversa si posible
        const v=DIRS[d], nc=this.tileC()+v.x, nr=this.tileR()+v.y;
        return !isWall(nc,nr);
      });
      if (options.length===0) { this.dir = OPP[this.dir]||'none'; return; }

      let target;
      if (this.mode==='fright'){
        // alejarnos de Pacman (target opuesto)
        const dx = this.tileC() - pac.tileC();
        const dy = this.tileR() - pac.tileR();
        target = {c: clamp(this.tileC()+Math.sign(dx)*6,1,COLS-2),
                  r: clamp(this.tileR()+Math.sign(dy)*6,1,ROWS-2)};
      } else if (this.mode==='eaten'){
        // volver a casa
        target = {c:GHOST_HOME.c, r:GHOST_HOME.r};
      } else {
        // mezcla: perseguir con 70% y dispersar 30%
        if (Math.random()<0.7) target = {c: pac.tileC(), r: pac.tileR()};
        else target = this.scatterTarget;
      }

      // Elegimos la opción que minimiza la distancia Manhattan al target (con un poco de ruido)
      let best = options[0], bestScore = Infinity;
      for (const d of options){
        const v=DIRS[d], nc=this.tileC()+v.x, nr=this.tileR()+v.y;
        const score = Math.abs(nc-target.c)+Math.abs(nr-target.r) + rand(0,0.6);
        if (score < bestScore){ bestScore=score; best=d; }
      }
      this.dir = best;
    }
    update(dt){
      this.chooseDir();
      this.move(dt, this.speed());
      // Si está comido y llega a casa, vuelve a 'chase'
      if (this.mode==='eaten' && this.homeReached()){
        this.setChase();
      }
      // Colisión con Pacman
      const dx = this.x - pac.x, dy = this.y - pac.y;
      if (dx*dx + dy*dy < (TILE*0.7)*(TILE*0.7)){
        if (this.mode==='fright'){
          this.setEaten(); score += 200;
        } else if (this.mode==='chase'){
          loseLife();
        }
      }
    }
    draw(){
      ctx.save();
      ctx.translate(this.x,this.y);
      if (this.eyesOnly){
        // ojos
        drawEyes();
      } else if (this.mode==='fright'){
        drawGhost('#3dd6ff');
      } else {
        drawGhost(this.color);
      }
      ctx.restore();

      function drawGhost(color){
        const w=TILE*0.95, h=TILE*0.95, r=h*0.52;
        ctx.fillStyle = color;
        ctx.beginPath();
        ctx.moveTo(-w*0.5, 0);
        ctx.lineTo(-w*0.5, -h*0.1);
        ctx.quadraticCurveTo(-w*0.5, -r, 0, -r);
        ctx.quadraticCurveTo(w*0.5, -r, w*0.5, -h*0.1);
        ctx.lineTo(w*0.5, 0);
        // ondas abajo
        const bumps=4, step=w/bumps;
        for (let i=0;i<bumps;i++){
          ctx.quadraticCurveTo(-w*0.5 + step*(i+0.5), h*0.35, -w*0.5 + step*(i+1), 0);
        }
        ctx.closePath();
        ctx.fill();
        drawEyes();
      }
      function drawEyes(){
        // ojos
        ctx.fillStyle='#fff';
        ctx.beginPath(); ctx.arc(-6,-6,4,0,Math.PI*2); ctx.fill();
        ctx.beginPath(); ctx.arc( 6,-6,4,0,Math.PI*2); ctx.fill();
        ctx.fillStyle='#1134ff';
        const v = DIRS[ghostFacing()];
        ctx.beginPath(); ctx.arc(-6+v.x*2,-6+v.y*2,2,0,Math.PI*2); ctx.fill();
        ctx.beginPath(); ctx.arc( 6+v.x*2,-6+v.y*2,2,0,Math.PI*2); ctx.fill();
      }
      function ghostFacing(){
        return (Math.abs(DIRSRef[0].x) || Math.abs(DIRSRef[0].y)) ? self.dir : 'left';
      }
    }
  }

  // pequeño truco para referenciar dirs en función local
  const DIRSRef=[DIRS.left]; // no usado pero evita tree-shake mental del motor

  // ====== Lógica de juego ======
  let pac = new Pacman();
  let ghosts = [0,1,2,3].map(i=> new Ghost(i));
  let dotsLeft = countDots();
  let score = 0;
  let level = 1;
  let lives = 3;
  let frightTimer = 0;
  let running = false;
  let last = 0;

  function isWall(c,r){
    return (r<0||r>=ROWS||c<0||c>=COLS) ? true : grid[r][c]===1;
  }

  function resetPositions(){
    pac = new Pacman();
    ghosts = [0,1,2,3].map(i=> new Ghost(i));
  }

  function nextLevel(){
    level++;
    levelEl.textContent = String(level);
    // repoblar puntos (sin cambiar muros)
    for (let r=0;r<ROWS;r++){
      for (let c=0;c<COLS;c++){
        if (grid[r][c]===0 && !(c===PAC_START.c&&r===PAC_START.r) && !(c===GHOST_HOME.c&&r===GHOST_HOME.r)){
          grid[r][c]=2;
        }
      }
    }
    POWER_POS.forEach(([c,r])=>{ if (grid[r]?.[c]===2 || grid[r]?.[c]===0) grid[r][c]=3; });
    dotsLeft = countDots();
    resetPositions();
  }

  function loseLife(){
    lives--;
    updateLives();
    if (lives<=0){
      running=false; showOverlay('💀 Juego terminado','Pulsa cualquier tecla para reiniciar');
    } else {
      resetPositions();
    }
  }

  function updateLives(){
    livesEl.innerHTML = '';
    for (let i=0;i<lives;i++){
      const d=document.createElement('div'); d.className='life'; livesEl.appendChild(d);
    }
  }

  function showOverlay(title,subtitle){
    stateTitle.textContent = title;
    stateSubtitle.textContent = subtitle;
    overlayMsg.hidden = false;
  }
  function hideOverlay(){ overlayMsg.hidden = true; }

  updateLives(); scoreEl.textContent='0'; levelEl.textContent='1';
  showOverlay('👋 ¡Listo!','Pulsa cualquier tecla para empezar');

  // ====== Entrada ======
  const keymap = { ArrowLeft:'left', ArrowRight:'right', ArrowUp:'up', ArrowDown:'down',
                   a:'left', d:'right', w:'up', s:'down', A:'left', D:'right', W:'up', S:'down' };
  window.addEventListener('keydown', (e)=>{
    if (!running){ running=true; hideOverlay(); last=performance.now(); return; }
    const d = keymap[e.key]; if (!d) return;
    pac.nextDir = d;
    e.preventDefault();
  }, {passive:false});

  // ====== Bucle principal ======
  function loop(ts){
    if (!running) { requestAnimationFrame(loop); return; }
    const dt = Math.min(0.035, (ts-last)/1000); // limitar delta para estabilidad
    last = ts;

    // Actualizar
    pac.update(dt);
    if (frightTimer>0){
      frightTimer-=dt;
      if (frightTimer<=0){ ghosts.forEach(g=>g.setChase()); frightTimer=0; }
    }
    ghosts.forEach(g=> g.update(dt));

    // Condición de pasar de nivel
    if (dotsLeft<=0){ nextLevel(); }

    // Pintar
    draw();

    requestAnimationFrame(loop);
  }

  function draw(){
    // Fondo
    ctx.clearRect(0,0,cvs.width,cvs.height);

    // Muros
    ctx.fillStyle = '#2039ff';
    for (let r=0;r<ROWS;r++){
      for (let c=0;c<COLS;c++){
        if (grid[r][c]===1){
          ctx.fillRect(c*TILE, r*TILE, TILE, TILE);
        }
      }
    }

    // Puntos
    for (let r=0;r<ROWS;r++){
      for (let c=0;c<COLS;c++){
        if (grid[r][c]===2){
          ctx.fillStyle = '#9ad0ff';
          ctx.beginPath();
          ctx.arc(c*TILE+TILE/2, r*TILE+TILE/2, 2.2, 0, Math.PI*2);
          ctx.fill();
        } else if (grid[r][c]===3){
          ctx.fillStyle = '#fff4a5';
          ctx.beginPath();
          ctx.arc(c*TILE+TILE/2, r*TILE+TILE/2, 4.0 + Math.sin(performance.now()/160)*0.8, 0, Math.PI*2);
          ctx.fill();
          // brillo
          ctx.globalAlpha = 0.35;
          ctx.beginPath(); ctx.arc(c*TILE+TILE/2, r*TILE+TILE/2, 8, 0, Math.PI*2); ctx.fillStyle='#fff4a5'; ctx.fill();
          ctx.globalAlpha = 1;
        }
      }
    }

    // Entidades
    pac.draw();
    ghosts.forEach(g=> g.draw());

    // HUD (score)
    scoreEl.textContent = String(score);
  }

  requestAnimationFrame(loop);

  })();
  </script>
